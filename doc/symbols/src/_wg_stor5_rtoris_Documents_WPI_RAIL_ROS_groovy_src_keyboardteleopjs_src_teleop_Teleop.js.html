  5 /**
  6  * Manages connection to the server and all interactions with ROS.
  7  *
  8  * Emits the following events:
  9  *  * 'change' - emitted with a change in speed occurs
 10  *
 11  * @constructor
 12  * @param options - possible keys include:
 13  *   * ros - the ROSLIB.Ros connection handle
 14  *   * topic (optional) - the Twist topic to publish to, like '/cmd_vel'
 15  *   * throttle (optional) - a constant throttle for the speed
 16  */
 17 KEYBOARDTELEOP.Teleop = function(options) {
 18   var that = this;
 19   var options = options || {};
 20   var ros = options.ros;
 21   var topic = options.topic || '/cmd_vel';
 22   // permanent throttle
 23   var throttle = options.throttle || 1.0;
 24 
 25   // used to externally throttle the speed (e.g., from a slider)
 26   this.scale = 1.0;
 27 
 28   // linear x and y movement and angular z movement
 29   var x = 0;
 30   var y = 0;
 31   var z = 0;
 32 
 33   var cmdVel = new ROSLIB.Topic({
 34     ros : ros,
 35     name : topic,
 36     messageType : 'geometry_msgs/Twist'
 37   });
 38 
 39   // sets up a key listener on the page used for keyboard teleoperation
 40   var handleKey = function(keyCode, keyDown) {
 41     // used to check for changes in speed
 42     var oldX = x;
 43     var oldY = y;
 44     var oldZ = z;
 45 
 46     var speed = 0;
 47     // throttle the speed by the slider and throttle constant
 48     if (keyDown === true) {
 49       speed = throttle * that.scale;
 50     }
 51     // check which key was pressed
 52     switch (keyCode) {
 53       case 65:
 54         // turn left
 55         z = 1 * speed;
 56         break;
 57       case 87:
 58         // up
 59         x = .5 * speed;
 60 
 61         break;
 62       case 68:
 63         // turn right
 64         z = -1 * speed;
 65         break;
 66       case 83:
 67         // down
 68         x = -.5 * speed;
 69         break;
 70       case 69:
 71         // strafe right
 72         y = -.5 * speed;
 73         break;
 74       case 81:
 75         // strafe left
 76         y = .5 * speed;
 77         break;
 78     }
 79 
 80     // publish the command
 81     var twist = new ROSLIB.Message({
 82       angular : {
 83         x : 0,
 84         y : 0,
 85         z : z
 86       },
 87       linear : {
 88         x : x,
 89         y : y,
 90         z : z
 91       }
 92     });
 93     cmdVel.publish(twist);
 94 
 95     // check for changes
 96     if (oldX !== x || oldY !== y || oldZ !== z) {
 97       that.emit('change', twist);
 98     }
 99   };
100 
101   // handle the key
102   var body = document.getElementsByTagName('body')[0];
103   body.addEventListener('keydown', function(e) {
104     handleKey(e.keyCode, true);
105   }, false);
106   body.addEventListener('keyup', function(e) {
107     handleKey(e.keyCode, false);
108   }, false);
109 };
110 KEYBOARDTELEOP.Teleop.prototype.__proto__ = EventEmitter2.prototype;
